43 inline std::size_t cl_preferred_work_size (std::size_t N,
44  const cl::Kernel &kern, const cl::Device &dev,
45  std::size_t &global_size, std::size_t &local_size)
46 {
47  std::size_t max_s;
48  std::size_t min_local;
49  try {
50  kern.getWorkGroupInfo(dev, CL_KERNEL_WORK_GROUP_SIZE, &max_s);
51  kern.getWorkGroupInfo(dev,
52  CL_KERNEL_PREFERRED_WORK_GROUP_SIZE_MULTIPLE, &min_local);
53  } catch (cl::Error) {
54  global_size = N;
55  local_size = 0;
56 
57  return 0;
58  }
59 
60  std::size_t max_local = (max_s / min_local) * min_local;
61  std::size_t mul_local = max_local / min_local;
62 
63  std::size_t pref_local = max_local;
64  std::size_t pref_global = cl_minimum_global_size(N, max_local);
65  std::size_t pref_diff = pref_global - N;
66 
67  for (std::size_t i = mul_local; i != 0; --i) {
68  std::size_t local = i * min_local;
69  std::size_t global = cl_minimum_global_size(N, local);
70  std::size_t diff = global - N;
71  if (diff < pref_diff) {
72  pref_local = local;
73  pref_global = global;
74  pref_diff = diff;
75  }
76  }
77 
78  global_size = pref_global;
79  local_size = pref_local;
80 
81  return pref_diff;
